    with open("my_lab_file", "r") as fh:
        print fh.read()

        fh.seek(0)
        for line in fh.readlines():
            print line

    with open("my_lab_file", "a") as fh:
        newlines = ["first new line\n", "second new line\n"]
        fh.writelines(newlines)
        print >> fh, "hi"
        print >> fh, "there"
