def product(a, b):
    """ Returns a * b. """
    return a * b


def only_odds(a):
    """ Returns True if a is an odd number. """
    if a % 2 == 1:
        return True
    return False 


def times_2(a, b):
    """ Returns (a + b) * 2. """
    return (a + b) * 2


if __name__ == "__main__":

    my_var = 9
